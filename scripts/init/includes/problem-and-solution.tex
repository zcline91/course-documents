% Commands for writing problems and solutions to different types
% of problems, e.g. for quizzes, tests, and worksheets.
\usepackage{import}

% Optional argument: space to leave for solution
% Required argument: path of problem
\newcommand{\problem}[2][]{%
  \question%
  \subimport{#2}{statement}%
  \ifblank{#1}{%
  }{%
    \begin{solutionbox}{#1}%
      \subimport{#2}{solution}%
    \end{solutionbox}%
  }%
}

% Same command as \problem, but using a part instead
% Optional argument: space to leave for solution
% Required argument: path of problem
\newcommand{\subproblem}[2][]{
  \part%
  \subimport{#2}{statement}%
  \ifblank{#1}{%
  }{%
    \begin{solutionbox}{#1}%
      \subimport{#2}{solution}%
    \end{solutionbox}%
  }%
}
