This is a solution to the problem.
\begin{align*}
  \defint x 0 2 {\left( x^2 - e^x \right)} 
    &= \left[\frac 1 3 x^3 - e^x\right]_0^2 \\
    &= \left( \frac 1 3 2^3 - e^2 \right) - \left( \frac 1 3 0^3 - e^0 \right) \\
    &= \frac 8 3 - e^2 + 1 \\
    &= \frac {11} 3 - e^2.
\end{align*}
