% Commands for writing problems and solutions to different types
% of problems, e.g. for quizzes, tests, and worksheets. There should
% be a command for every problem source.
\usepackage{import}

% Optional argument: space to leave for solution
% Required argument: path of problem
\newcommand{\problem}[2][]{%
  \question%
  \subimport{#2}{statement}%
  \ifblank{#1}{%
  }{%
    \begin{solutionbox}{#1}%
      \subimport{#2}{solution}%
    \end{solutionbox}%
  }%
}

% Same command as \problem, but using a part instead
% Optional argument: space to leave for solution
% Required argument: path of problem
\newcommand{\subproblem}[2][]{
  \part%
  \subimport{#2}{statement}%
  \ifblank{#1}{%
  }{%
    \begin{solutionbox}{#1}%
      \subimport{#2}{solution}%
    \end{solutionbox}%
  }%
}

\newcommand{\beslproblem}[4][]{%
  \quizproblem[#1]{../../../problems/bes12b/#2/#3/#4}
}
\newcommand{\beslsubproblem}[5][]{%
  \quizpart[#1]{../../../problems/bes12b/#2/#3/#4/#5}
}

\newcommand{\assessmentproblem}[2][]{%
  \quizproblem[#1]{../../../problems/assessment/#2}
}
\newcommand{\assessmentsubproblem}[3][]{%
  \quizpart[#1]{../../../problems/assessment/#2/#3}
}
